  <style>

    body {

    }

  </style>

</head>

<body class="bg-gray-100 min-h-screen flex items-center justify-center p-4">

  <div class="bg-white shadow-xl rounded-2xl p-6 w-full max-w-xl">

    <h1 class="text-2xl font-semibold text-center text-gray-800 mb-4">🗓️ Daily Schedule Planner</h1>

    <div class="flex gap-2 mb-4">

      <input type="time" id="task-time" class="w-1/3 p-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-400" />

      <input type="text" id="task-desc" placeholder="What do you need to do?" class="w-2/3 p-2 border border-gray-300 rounded-lg focus:outline-none focus:ring-2 focus:ring-blue-400" />

    </div>

    <button onclick="addTask()" class="w-full bg-blue-600 hover:bg-blue-700 text-white font-semibold py-2 rounded-lg mb-6 transition">Add Task</button>

    <div id="task-list" class="space-y-3"></div>

  </div>

  <script>

    let tasks = JSON.parse(localStorage.getItem('tasks')) || [];

    function saveTasks() {

      localStorage.setItem('tasks', JSON.stringify(tasks));

    }

    function renderTasks() {

      const list = document.getElementById('task-list');

      list.innerHTML = '';

      tasks.forEach((task, index) => {

        const div = document.createElement('div');

        div.className = `flex items-center justify-between p-3 rounded-lg border ${task.done ? 'bg-green-100 border-green-300' : 'bg-gray-50 border-gray-200'}`;

        div.innerHTML = `

          <div class="flex items-center gap-3">

            <input type="checkbox" ${task.done ? 'checked' : ''} onclick="toggleDone(${index})" class="w-5 h-5 text-blue-600 rounded">

            <div>

              <div class="text-sm text-gray-500">${task.time}</div>

              <div class="font-medium ${task.done ? 'line-through text-gray-400' : 'text-gray-800'}">${task.desc}</div>

            </div>

          </div>

          <div class="flex gap-2">

            <button onclick="editTask(${index})" class="text-green-600 hover:text-green-800 text-sm">✏️</button>

            <button onclick="deleteTask(${index})" class="text-red-500 hover:text-red-700 text-sm">🗑️</button>

          </div>

        `;

        list.appendChild(div);

      });

    }

    function addTask() {

      const time = document.getElementById('task-time').value;

      const desc = document.getElementById('task-desc').value.trim();

      if (!time || !desc) {

        alert("Please enter both time and task.");

        return;

      }

      tasks.push({ time, desc, done: false });

      saveTasks();

      renderTasks();

      document.getElementById('task-time').value = '';

      document.getElementById('task-desc').value = '';

    }

    function toggleDone(index) {

      tasks[index].done = !tasks[index].done;

      saveTasks();

      renderTasks();

    }

    function deleteTask(index) {

      if (confirm("Delete this task?")) {

        tasks.splice(index, 1);

        saveTasks();

        renderTasks();

      }

    }

    function editTask(index) {

      const newTime = prompt("Edit time:", tasks[index].time);

      const newDesc = prompt("Edit description:", tasks[index].desc);

      if (newTime && newDesc) {

        tasks[index].time = newTime;

        tasks[index].desc = newDesc;

        saveTasks();

        renderTasks();

      }

    }

    renderTasks();

  </script>
